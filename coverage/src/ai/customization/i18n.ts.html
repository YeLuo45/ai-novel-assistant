export function t(key: string, lang: Language = 'zh-CN'): string {
  const keys = key.split('.')
  let value: any = TRANSLATIONS[lang]
  
  for (const k of keys) {
    if (value && typeof value === 'object' && k in value) {
      value = value[k]
    } else {
      return key
    }
  }
  
  return typeof value === 'string' ? value : key
}
